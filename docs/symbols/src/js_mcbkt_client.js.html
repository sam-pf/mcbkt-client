1 /**
  2  * Makes an ajax call to url and returns a Promise object.
  3  * @param {string} url - URL to connect to.
  4  * @param {string} method - Currently, accepted values are 'get' and 'post'
  5  *   (case-independent).  This argument is meaningful only if url uses
  6  *   http(s) protocol.
  7  * @param {JSON.stringfy-able} data - Any data that will be sent as JSON with
  8  *   the URL request.  If data is passed a value, then "Content-Type" header
  9  *   is set as JSON for the XMLHttpRequest.  JSON.stringify is called on data
 10  *   in order to prepare JSON to send.
 11  * @return A promise instance.  The returned object can be used as follows:
 12  *   <pre>
 13  *     ajax_as_promise ('https://...')
 14  *     .then ((data) => {
 15  *       // Do stuff with data, if foo.txt was successfully loaded.
 16  *     })
 17  *     .catch ((err) => {
 18  *       // Do stuff on error...
 19  *     })
 20  *   </pre>
 21  * @see https://stackoverflow.com/questions/8567114/
 22  */
 23 var ajax_as_promise = function (url, method = "GET", data) {
 24   method = method.toUpperCase ()
 25   if (! /^(GET|POST)$/.test (method))
 26     throw Error (`Unsupported method: ${method}`)
 27   return new Promise ((resolve, reject) => {
 28     const req = new XMLHttpRequest ()
 29     req.open (method, url)
 30     req.onload = () => req.status === 200 ? resolve (req.response) :
 31       reject (Error (req.statusText))
 32     req.onerror = (e) => reject (Error (`Network Error: ${e}`))
 33     if ( data != undefined ) {
 34       req.setRequestHeader ("Content-Type", "application/json;charset=UTF-8")
 35       data = JSON.stringify (data)
 36     }
 37     req.send (data)
 38   })
 39 }
